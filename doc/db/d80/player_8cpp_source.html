139 void PlayerInterface::checkPlaybackStatusChange(SongInfo &song)
140 {
141  PlaybackStatus status = Stopped;
142  if(song.isValid())
143  status = song.playbackStatus;
144  if(status != this->currentStatus)
145  {
146  qDebug() << "Playback status changed: "
147  << playbackStatusToString((status));
148  this->currentStatus = status;
149  emit playbackStatusChanged(status);
150  }
151 }

153 //---------------------------------------------------------
154 void PlayerInterface::checkSongChange(SongInfo &song)
155 {
156  if(song.songName != this->_currentSong.songName &&
157  song.playbackStatus == Playing)
158  {
159  qDebug() << "Song changed: " << song.songName;
160  this->_currentSong = song;
161  emit songChanged(song);
162  if(song.isValid())
163  this->downloadSongArt(song.artUrl);
164  }
165 }

178 void PlayerInterface::downloadSongArt(const QString& url)
179 {
180  UrlDownloader* downloader = new UrlDownloader(this);
181  QString dest = QDir::temp().path() +
182  QString(QDir::separator()) +
183  this->currentSong().songId +
184  QFileInfo(url).fileName();

186  this, SLOT(onArtReady(const QString&)));

188  downloader->download(url, dest);
