35 void ExtensionsManager::loadPlugin(IExtension* iExtension)
36 {
37  if(!plugin(iExtension->metaData().name))
38  {
39  m_plugins.append(iExtension);
40  qDebug() << "Extension plugin loaded: ";
41  qDebug() << " - name: " << iExtension->metaData().name;
42  qDebug() << " - version: " << iExtension->metaData().version;
43  if(QSettings().value(iExtension->metaData().name, true).toBool())
44  iExtension->setup();
