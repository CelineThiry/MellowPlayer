80 bool StreamingServicesManager::startService(const QString& serviceName) {
81  bool retVal = true;
82  IStreamingService* p = plugin(serviceName);
83  if(!p)
84  retVal = false;
85  else
86  {
87  if(!Services::hasFlash() && p->flashRequired())
88  {
89  QMessageBox::warning(
90  Services::mainWindow(),
91  tr("Flash player plugin missing"),
92  tr("The flash player plugin is missing on your system.\n\n"
93  "This service cannot be used without flash.\nPlease "
94  "install the latest flash player plugin from adobe and try "
95  "again!"));
96  retVal = false;
97  }
98  else if(p != m_currentService)
99  {
100  m_currentService = p;
101  qDebug() << "Starting service " << serviceName
102  << "(" << p->url() << ")";
103  Services::webView()->load(m_currentService->url());
104  retVal = true;
105  }
106  }
107  return retVal;
108 }
